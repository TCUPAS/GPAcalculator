<script>
const PREREQS = [
{id:'engl', name:'English Composition', defaultCr:3, hasLab:false, allowAP:true},
{id:'micro', name:'Microbiology', defaultCr:3, hasLab:true, labName:'Microbiology Lab', labCr:1},
{id:'biochem',name:'Biochemistry', defaultCr:3, hasLab:false},
{id:'genetics',name:'Genetics', defaultCr:3, hasLab:false},
{id:'chem1', name:'Chemistry I', defaultCr:3, hasLab:true, labName:'Chemistry I Lab', labCr:1},
{id:'chem2', name:'Chemistry II', defaultCr:3, hasLab:true, labName:'Chemistry II Lab', labCr:1},
{id:'ochem', name:'Organic Chemistry I', defaultCr:3, hasLab:true, labName:'Organic Chemistry I Lab',labCr:1},
{id:'anat', name:'Human Anatomy (or A&P 1)', defaultCr:3, hasLab:true, labName:'Anatomy Lab', labCr:1},
{id:'physio', name:'Human Physiology (or A&P 2)', defaultCr:3, hasLab:true, labName:'Physiology Lab', labCr:1},
{id:'psyc', name:'Psychology', defaultCr:3, hasLab:false},
{id:'stats', name:'Statistics', defaultCr:3, hasLab:false},
{id:'physics',name:'Physics', defaultCr:3, hasLab:true, labName:'Physics Lab', labCr:1},
{id:'medterm',name:'Medical Terminology', defaultCr:1, hasLab:false},
];

const GRADES = ['A+','A','A-','B+','B','B-','C+','C','C-','D+','D','D-','F'];

function gradePoints(g) {
if (['A+','A','A-'].includes(g)) return 4.0;
if (['B+','B','B-'].includes(g)) return 3.0;
if (['C+','C','C-'].includes(g)) return 2.0;
if (['D+','D','D-'].includes(g)) return 1.0;
return 0.0;
}

function isBelowC(g) { return ['C-','D+','D','D-','F'].includes(g); }

const state = {};
PREREQS.forEach(p => {
state[p.id] = { attempts: [{grade:'A', cr:p.defaultCr}], splitLab: false, apCredit: false };
if (p.hasLab) state[p.id+'_lab'] = { attempts: [{grade:'A', cr:p.labCr}] };
});

function gradeSelectHTML(cid, idx, val) {
const opts = GRADES.map(g => `<option value="${g}"${g===val?' selected':''}>${g}</option>`).join('');
return `<select onchange="updateGrade('${cid}',${idx},this.value)">${opts}</select>`;
}

function renderAttempts(cid, attempts) {
return attempts.map((a, i) => `
<div class="attempt-block">
<div class="attempt-row">
<div>
<div class="attempt-label">Attempt ${i+1}</div>
<div class="cr-row">
<span>Credits:</span>
<input type="number" value="${a.cr}" min="0" max="12" step="0.5"
onchange="updateCr('${cid}',${i},this.value)">
</div>
</div>
${gradeSelectHTML(cid, i, a.grade)}
<button class="remove-btn" onclick="removeAttempt('${cid}',${i})" aria-label="Remove attempt">
<i class="ti ti-x" style="font-size:14px"></i>
</button>
</div>
${isBelowC(a.grade) ? `<div class="warn-badge"><i class="ti ti-alert-triangle" style="font-size:11px"></i> Below C — does not meet minimum</div>` : ''}
</div>
`).join('');
}

function renderAll() {
const container = document.getElementById('courses-container');
let html = '';

PREREQS.forEach(p => {
const cs = state[p.id];
const splitLab = p.hasLab && cs.splitLab;
const labState = p.hasLab ? state[p.id+'_lab'] : null;

html += `<div class="section-card">`;
html += `<div class="course-name">${p.name}</div>`;

if (p.allowAP) {
const apChecked = cs.apCredit;
html += `<label class="lab-toggle" style="margin-bottom:${apChecked?'0':'10px'}">
<input type="checkbox" ${apChecked?'checked':''} onchange="toggleAP('${p.id}',this.checked)">
Fulfilled via AP credit
</label>`;
if (apChecked) {
html += `<div style="font-size:12px;color:var(--text-secondary);margin-top:6px;padding:6px 10px;background:var(--bg-secondary);border-radius:var(--radius-md)">AP credit accepted for English — excluded from GPA calculation.</div>`;
html += `</div>`;
return;
}
}

if (p.hasLab) {
html += `<label class="lab-toggle">
<input type="checkbox" ${splitLab?'checked':''} onchange="toggleLab('${p.id}',this.checked)">
Lab was graded separately
</label>`;
}

html += renderAttempts(p.id, cs.attempts);
html += `<button class="add-attempt" onclick="addAttempt('${p.id}')">
<i class="ti ti-plus" style="font-size:13px"></i> Add attempt
</button>`;

if (splitLab && labState) {
html += `<div class="lab-divider">`;
html += `<div class="lab-section-title">${p.labName}</div>`;
html += renderAttempts(p.id+'_lab', labState.attempts);
html += `<button class="add-attempt" onclick="addAttempt('${p.id}_lab')">
<i class="ti ti-plus" style="font-size:13px"></i> Add attempt
</button>`;
html += `</div>`;
}

html += `</div>`;
});

container.innerHTML = html;
recalc();
}

function toggleAP(pid, val) {
state[pid].apCredit = val;
renderAll();
}

function toggleLab(pid, val) {
state[pid].splitLab = val;
renderAll();
}

function addAttempt(cid) {
const p = PREREQS.find(x => x.id===cid || x.id+'_lab'===cid);
const defaultCr = cid.endsWith('_lab') ? (p ? p.labCr : 1) : (p ? p.defaultCr : 3);
state[cid].attempts.push({grade:'A', cr:defaultCr});
renderAll();
}

function removeAttempt(cid, i) {
if (state[cid].attempts.length <= 1) return;
state[cid].attempts.splice(i, 1);
renderAll();
}

function updateGrade(cid, i, val) {
state[cid].attempts[i].grade = val;
recalc();
const container = document.getElementById('courses-container');
const rows = container.querySelectorAll('.attempt-block');
}

function updateCr(cid, i, val) {
state[cid].attempts[i].cr = parseFloat(val) || 0;
recalc();
}

function recalc() {
let totalPoints = 0, totalCr = 0, totalAttempts = 0, warnings = 0;

const allCids = [];
PREREQS.forEach(p => {
if (state[p.id].apCredit) return;
allCids.push(p.id);
if (p.hasLab && state[p.id].splitLab) allCids.push(p.id+'_lab');
});

allCids.forEach(cid => {
const cs = state[cid];
if (!cs) return;
cs.attempts.forEach(a => {
if (a.cr > 0) {
totalPoints += gradePoints(a.grade) * a.cr;
totalCr += a.cr;
totalAttempts++;
if (isBelowC(a.grade)) warnings++;
}
});
});

const gpa = totalCr > 0 ? totalPoints / totalCr : null;
document.getElementById('gpa-out').textContent = gpa !== null ? gpa.toFixed(2) : '—';
}

renderAll();
</script>
